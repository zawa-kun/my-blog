Python コードブロック

# Pythonでのクラス定義
class BlogPost:
    def __init__(self, title, content):
        self.title = title
        self.content = content
        self.published = False

    def publish(self):
        self.published = True
        print(f"記事「{self.title}」を公開しました！")
        return True

# インスタンス作成と使用
post = BlogPost("マークダウンの書き方", "マークダウンは簡単に文書を作成できます。")
post.publish()

JavaScript コードブロック

// ReactコンポーネントのJavaScript例
const BlogCard = ({ title, excerpt, date }) => {
  const formattedDate = new Date(date).toLocaleDateString("ja-JP");

  return (
    <div className="blog-card">
      <h3>{title}</h3>
      <time dateTime={date}>{formattedDate}</time>
      <p>{excerpt}</p>
      <button onClick={() => console.log("クリックされました")}>
        続きを読む
      </button>
    </div>
  );
};

export default BlogCard;
